TConflictRecord is a record for holding information about a conflict that has occured.

TConflictRecord = record
  Node1: string;
  Node2: string;
  ChangedFields1: TStringList;
  ChangedFields2: TStringList;
  ConflictingFields: TStringList;
  FirstOperationDate1: TDateTime;
  FirstOperationDate2: TDateTime;
  LastOperationDate1: TDateTime;
  LastOperationDate2: TDateTime;
  NumChanges1: Integer;
  NumChanges2: Integer;
  ChosenNode: string;
  end;
struct TConflictRecord {
  string Node1;
  string Node2;
  TStringList ChangedFields1;
  TStringList ChangedFields2;
  TStringList ConflictingFields;
  TDateTime FirstOperationDate1;
  TDateTime FirstOperationDate2;
  TDateTime LastOperationDate1;
  TDateTime LastOperationDate2;
  int NumChanges1;
  int NumChanges2;
  string ChosenNode;
};
